This inspection reports when more than one default source is set inside the @Mapping 


//wrong
@Mapper
public interface EmployeeMapper {
     @Mapping(source = "employeeName", target = "name", defaultValue = "My name", defaultExpression = "java(UUID.randomUUID().toString())")
    Employee toEmployee(EmployeeDto employeeDto, @Context CycleAvoidingMappingContext context);
}



//correct
@Mapper
public interface EmployeeMapper {
    @Mapping(source = "employeeName", target = "name",  defaultValue = "My name")
    Employee toEmployee(EmployeeDto employeeDto, @Context CycleAvoidingMappingContext context);
}
